def count(arr):
  count_danji = []
  for i in range(len(arr)):
    for j in range(len(arr)):
      if arr[i][j] == 1:
        count_danji.append(1)
        arr[i][j] = 0
        count_danji[len(count_danji)-1] += near(arr, i, j)

  return count_danji;

def near(arr, i, j):
  near_count = 0;

  if j-1 >= 0 and arr[i][j-1] == 1:
    near_count += 1
    arr[i][j-1] = 0
    near_count += near(arr, i, j-1)

  if i-1 >= 0 and arr[i-1][j] == 1:
    near_count += 1
    arr[i-1][j] = 0
    near_count += near(arr, i-1, j)

  if j+1 < len(arr) and arr[i][j+1] == 1:
    near_count += 1
    arr[i][j+1] = 0
    near_count += near(arr, i, j+1)

  if i+1 < len(arr) and arr[i+1][j] == 1:
    near_count += 1
    arr[i+1][j] = 0
    near_count += near(arr, i+1, j)

  return near_count;


if __name__ == "__main__":
  n = int(input())

  arr = [[0 for i in range(n)] for j in range(n)]

    line = input()

      arr[i][idx] = int(num)
